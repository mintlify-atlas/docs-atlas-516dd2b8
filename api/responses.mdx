## Overview

Response types represent messages sent from agents to clients in reply to requests. Each response corresponds to a specific request type.

## Initialize

### InitializeResponse

Response to an initialization request.

<ResponseField name="protocolVersion" type="int" required>
  The protocol version the agent will use
</ResponseField>

<ResponseField name="agentInfo" type="Implementation">
  Information about the agent implementation
  
  <Expandable title="Implementation fields">
    <ResponseField name="name" type="string" required>
      The name of the implementation
    </ResponseField>
    
    <ResponseField name="version" type="string" required>
      The version of the implementation
    </ResponseField>
    
    <ResponseField name="title" type="string">
      A human-readable title for the implementation
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="agentCapabilities" type="AgentCapabilities">
  Capabilities advertised by the agent
  
  <Expandable title="AgentCapabilities fields">
    <ResponseField name="loadSession" type="bool" default="false">
      Whether the agent supports loading existing sessions
    </ResponseField>
    
    <ResponseField name="sessionCapabilities" type="SessionCapabilities">
      Session-related capabilities
      
      <Expandable title="SessionCapabilities fields">
        <ResponseField name="fork" type="SessionForkCapabilities">
          Whether session forking is supported
        </ResponseField>
        
        <ResponseField name="list" type="SessionListCapabilities">
          Whether listing sessions is supported
        </ResponseField>
        
        <ResponseField name="resume" type="SessionResumeCapabilities">
          Whether resuming sessions is supported
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="mcpCapabilities" type="McpCapabilities">
      MCP server capabilities
      
      <Expandable title="McpCapabilities fields">
        <ResponseField name="http" type="bool" default="false">
          Whether HTTP MCP servers are supported
        </ResponseField>
        
        <ResponseField name="sse" type="bool" default="false">
          Whether SSE MCP servers are supported
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="promptCapabilities" type="PromptCapabilities">
      Prompt-related capabilities
      
      <Expandable title="PromptCapabilities fields">
        <ResponseField name="audio" type="bool" default="false">
          Whether audio content is supported in prompts
        </ResponseField>
        
        <ResponseField name="image" type="bool" default="false">
          Whether image content is supported in prompts
        </ResponseField>
        
        <ResponseField name="embeddedContext" type="bool" default="false">
          Whether embedded context is supported
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="authMethods" type="List<AuthMethod>" default="[]">
  Available authentication methods
  
  <Expandable title="AuthMethod fields">
    <ResponseField name="id" type="String" required>
      Unique identifier for the auth method
    </ResponseField>
    
    <ResponseField name="name" type="String" required>
      Display name for the auth method
    </ResponseField>
    
    <ResponseField name="description" type="String">
      Description of the auth method
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="_meta" type="Map<String, dynamic>">
  Optional metadata
</ResponseField>

## Prompt

### PromptResponse

Response to a prompt request. See [Stop Reasons](https://agentclientprotocol.com/protocol/prompt-turn#stop-reasons).

<ResponseField name="stopReason" type="StopReason" required>
  Indicates why the agent stopped processing the turn:
  - `end_turn` - The turn ended successfully
  - `max_tokens` - Reached the maximum number of tokens
  - `max_turn_requests` - Reached the maximum number of allowed agent requests
  - `refusal` - The agent refused to continue
  - `cancelled` - The turn was cancelled via `session/cancel`
</ResponseField>

<ResponseField name="usage" type="Usage">
  Token usage information for this prompt
  
  <Expandable title="Usage fields">
    <ResponseField name="inputTokens" type="int" required>
      Number of input tokens consumed
    </ResponseField>
    
    <ResponseField name="outputTokens" type="int" required>
      Number of output tokens generated
    </ResponseField>
    
    <ResponseField name="totalTokens" type="int" required>
      Total tokens (input + output)
    </ResponseField>
    
    <ResponseField name="cachedReadTokens" type="int">
      Number of cached tokens read
    </ResponseField>
    
    <ResponseField name="cachedWriteTokens" type="int">
      Number of tokens written to cache
    </ResponseField>
    
    <ResponseField name="thoughtTokens" type="int">
      Number of tokens used for internal reasoning
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="_meta" type="Map<String, dynamic>">
  Optional metadata
</ResponseField>
